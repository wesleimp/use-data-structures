# useBubbleSort

```js
import { useBubbleSort } from "use-data-structures"

function App() {
  const unsortedArray = [64, 25, 12, 22, 11];
  const sortedArray = useBubbleSort(unsortedArray);
  // you can pass a custom predicate as the second argument
  // to change how the array will be sorted
  // (e.g. (a, b) => a > b) to sort the list in descending order

  return (
    <div>
      <h2>Unsorted array</h2>
      <p>{ unsortedArray.join(",").toString() }</p>

      <h2>Sorted array</h2>
      <p>{ sortedArray.join(",").toString() }</p>
    </div>
  );
}
```
